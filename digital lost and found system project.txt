# Lists to store the items
lost_items = []
found_items = []

def add_lost_item():
    item = input("Enter the name of the lost item: ").lower()
    lost_items.append(item)
    print(f"Success: '{item}' has been added to the Lost list.")

def add_found_item():
    item = input("Enter the name of the found item: ").lower()
    found_items.append(item)
    print(f"Success: '{item}' has been added to the Found list.")

def match_items():
    # Finding common items between both lists
    matches = set(lost_items) & set(found_items)
    
    print("\n--- Matching Results ---")
    if matches:
        print(f"Match Found! The following items are in both lists: {', '.join(matches)}")
    else:
        print("No matches found yet. Keep looking!")
    print("------------------------")

def main():
    while True:
        print("\n1. Report Lost Item")
        print("2. Report Found Item")
        print("3. Match Items")
        print("4. Exit")
        
        choice = input("Enter choice: ")
        
        if choice == "1":
            add_lost_item()
        elif choice == "2":
            add_found_item()
        elif choice == "3":
            match_items()
        elif choice == "4":
            print("Exiting the system. Goodbye!")
            break
        else:
            print("Invalid choice! Please enter a number between 1 and 4.")

if __name__ == "__main__":
    main()
