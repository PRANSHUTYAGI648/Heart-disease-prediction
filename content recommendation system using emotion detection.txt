from textblob import TextBlob

# 1. Database of Content
# In a real app, this could be a CSV file or a SQL database.
content_db = {
    "happy": ["Upbeat Pop Playlist 🎵", "Comedy Special: John Mulaney 🎬", "Outdoor Adventure Vlog 🧗"],
    "sad": ["Lo-fi Chill Beats 🎧", "Inspirational Ted Talk 💡", "Comfort Food Cooking Video 🍲"],
    "angry": ["Guided Meditation for Calm 🧘", "Boxing Workout Routine 🥊", "Nature Sounds: Rain & Thunder 🌧️"],
    "neutral": ["Daily Tech News 💻", "Educational Science Podcast 🎙️", "Minimalist Room Tour 🏠"],
    "excited": ["High-Energy EDM Mix 🔥", "Fast-Paced Action Movie Trailer 🏎️", "Startup Brainstorming Tips 🚀"]
}

# 2. Hybrid Emotion Detection Function
def detect_emotion(user_text):
    user_text = user_text.lower().strip()
    
    # Priority 1: Keyword Matching (Catching strong emotions quickly)
    keywords = {
        "angry": ["kill", "hate", "angry", "furious", "mad", "annoyed"],
        "excited": ["wow", "awesome", "yay", "excited", "can't wait", "amazing"],
        "sad": ["cry", "lonely", "depressed", "hurt", "unhappy"]
    }
    
    for emotion, words in keywords.items():
        if any(word in user_text for word in words):
            return emotion

    # Priority 2: Sentiment Analysis (For general sentences)
    analysis = TextBlob(user_text)
    polarity = analysis.sentiment.polarity # Ranges from -1 to 1
    
    if polarity > 0.1:
        return "happy"
    elif polarity < -0.1:
        return "sad"
    else:
        return "neutral"

# 3. Recommendation Engine
def get_recommendations(emotion):
    # Default to neutral if emotion is not found
    return content_db.get(emotion, content_db["neutral"])

# 4. Main Application Loop
def main():
    print("============================================")
    print("   AI EMOTION CONTENT RECOMMENDER v1.0")
    print("============================================")
    print("(Type 'quit' to exit the program)\n")

    while True:
        user_input = input("How are you feeling right now? ")

        if user_input.lower() in ['quit', 'exit', 'bye']:
            print("Goodbye! Stay mindful. ✨")
            break

        # Step 1: Detect
        detected_emotion = detect_emotion(user_input)
        print(f"\n>> Detected Mood: {detected_emotion.upper()}")

        # Step 2: Recommend
        suggestions = get_recommendations(detected_emotion)
        
        print(">> Recommended for you:")
        for i, item in enumerate(suggestions, 1):
            print(f"   {i}. {item}")
        print("-" * 40)

if __name__ == "__main__":
    main()
